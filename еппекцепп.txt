ls (list) — список всех файлов в папке, в которой находимся

cd (change directory) — переход в нужную папку

git --version — вывести версию Git

clear (с англ. очистить) — очистить экран консоли

git init — создает новый репозиторий Git в текущей папке

cd .git & ls — перейти в директорию с настройками Git и вывести названия файлов и директорий

cd .. — вернуться на директорию выше

git status — отображает текущее состояние репозитория - показывает состояния файлов в рабочей директории и индексе: какие файлы изменены, но не добавлены в индекс; какие ожидают коммита в индексе.

git add <нужный файл> — добавить измененный/новый файл (индексирует) в последующие изменения (коммит)

git rm --cached <нужный файл> — удалить файл из текущей индексации перед коммитом

git add . — добавить все файлы в текущей папке в последующие изменения (коммит)

git commit -m "<ваше сообщение с описанием>" — фиксирует проиндексированные изменения и сохраняет их, а также добавляет описание изменения

git branch — посмотреть основную ветку (выделена значком * и все ветки)

git branch <название ветки> — создать новую ветку

git branch -d <название ветки> — удалить выбранную ветку

git checkout <название ветки> — переключится на ветку

git checkout -b <название ветки> — создать новую ветку и сразу переключиться

git merge <название ветки> — объединить изменения текущей ветки с выбранной
